About

Sharpitor Gardens is a library extension for the programming language Processing. It provides a lightweight framework for infinite-extent world creation and navigation.

Features

  • frozen capillary wave approach to infinite terrain generation
  • fixed spatial density maintenance of plant/animal species
  • first person navigation perspective with obstacle handling
  • machinima-style recording/playback
  • L-system species (algorithmically generated trees)

    • Tested

    Platform Ubuntu (Linux)
    Processing 1.0.1
    Dependencies Processing core

/***************************************************************************

    Sharpitor Gardens Demo -
	
    	creates a North Norwegian landscape with a joystick for navigation  

****************************************************************************/

import com.sharpitor.tellus.*;		     // base class package for unpopulated world comprising terrain, lakes, sky
import com.sharpitor.tellus.tellurians.*;    // class package for adding actor/obstacle populations to the world (plants, animals)
import com.sharpitor.tellus.lindemayer.*;    // subclass package for L-system tellurians (trees)

import com.sharpitor.dashboard.*;	    // class package for interactive navigation dashboard

// ---------------------------------------------------

Tellus theWorld;   		// parent world instance to which we will add terrain, sky and trees
Dashboard theDashboard;		// dashboard instance for navigation

boolean playback=false;		// make the world interactive

public void setup() {		// initialization
  size(400, 300);  
  frameRate(30);  
  
  theWorld = new Tellus(this,playback); 	      // create the world; empty by default
  
  theWorld.createTerrain(20,180,color(200,160,140));  // parameters specify steepness, colour variation 
  theWorld.createSky(color(230,250,255),600); 	      // parameters specify colour, horizon offset 
  theWorld.createForest(5,1); 			      // sparsely populate with trees; 
						      // parameters specify population density, tree type 
 
  
  theDashboard = new Dashboard(this); 		// create dashboard; empty by default
  theDashboard.addJoystick(330,240,0.8); 	// parameters specify position and size on the dashboard
  theDashboard.addCompass(250,240,0.8);
  theWorld.interfaceWith(theDashboard);		// connect world view response to the dashboard controls

}

public void draw() {		// interactive animation/navigation loop

 theDashboard.update();
 theWorld.update();
 
 theWorld.draw();
 theDashboard.draw();
}

// event handlers

public void mousePressed(){
  theDashboard.mousePressed();
}

public void mouseReleased(){
 theDashboard.mouseReleased();  
}
